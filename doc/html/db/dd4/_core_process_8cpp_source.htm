182  CrazyCrush::Game::Control::levelWin();

186  }

193  int topPos;
194  irr::core::position2d<irr::s32> pos;

202  continue;

204  Screen::matrix.at(i,j)->setDisplay(true);
205  continue;
206  }
207  isTop = false;
208  for(unsigned i2=i+1;i2<Screen::num_rows;i2++){
209  if((!Screen::matrix.at(i2,j)->getDisplay())||
210  (Screen::matrix.at(i2,j)->getFixed()))
211  continue;
212  Screen::unselectBlock(i2,j);
213  someoneFalled = true;
214  isTop = true;
215  sw = Screen::matrix.at(i,j);
216  Screen::matrix.at(i,j) = Screen::matrix.at(i2,j);
217  Screen::matrix.at(i2,j) = sw;
218  Screen::matrix.at(i,j)->setEnabled(false);
219  MorphLinear *t = new MorphLinear();
220  t->setFinalPosition(Screen::convertMatrix2Screen(i,j));
221  t->setVelocity(0.0,Screen::fall_velocity);
222  Screen::matrix.at(i,j)->appendMorph(t).startAnimator(eTime).appendCallback(new CallbackAfterFall(i,j));
223  Screen::AnimationCount::increase();
224  break;
225  }
226  if(!isTop){
227  someoneFalled = true;
228  topPos -= TextureCache::gemHeight;
229  Screen::matrix.at(i,j)->setScreenPosition(pos.X,topPos);
230  MorphLinear *t = new MorphLinear();
231  t->setFinalPosition(Screen::convertMatrix2Screen(i,j));
232  t->setVelocity(0.0,Screen::fall_velocity);
233  Screen::matrix.at(i,j)->setEnabled(false);
234  Screen::matrix.at(i,j)->randGem()
235  .appendMorph(t).startAnimator(eTime).setDisplay(true).appendCallback(new CallbackAfterFall(i,j));
236  Screen::AnimationCount::increase();
237  }
238  }
239  }
240  return someoneFalled;
