484 void showPage(){
485  std::stringstream SS;
486  std::string S;
487  Callback *x = NULL;
488  switch(page){
489  default:
490  case 1:
491  description->setString("Click over one gem to select it.");
492  x = new CallbackAnimationPage1();
493  x->process();
494  break;
495  case 2:
496  description->setString("Click on gem next to selected gem to swap then.\nDiagonal isn't possible.");
497  x = new CallbackAnimationPage2();
498  x->process();
499  break;
500  case 3:
501  description->setString("Swap gems to make 3-in-row or 3-in-column identical gems\nand destroy them.\nNote: 3 or more. Maximum possible is 7 gems simultaneous.");
502  x = new CallbackAnimationPage3();
503  x->process();
504  break;
505  case 4:
506  description->setString("In Level mode, change the background color destroying a gem \nabove it. Change all colors to transparency to win the level.\nColor sequence: Red -> Blue -> Green -> transparency");
507  x = new CallbackAnimationPage4();
508  x->process();
509  break;
510  case 5:
511  description->setString("Blocks are fixed. You can't move them and can't move the gems \nthrough them.");
512  x = new CallbackAnimationPage5();
513  x->process();
514  break;
515  case 6:
516  description->setString("Enjoy");
517  break;
518  }
519  if(page != 6)
520  screen->setVisibility(s,false);
521  else
522  screen->setVisibility(s,true);
523  if(x != NULL)
524  deleteCallback(x);
525 
526  S="";
527  SS.str(S);
528  SS << "Page " << page << " of " << totalPages;
529  lblPage->setString(SS.str().c_str());
530  if(page == 1){
531  screen->setVisibility(btnBack,false);
532  }else{
533  screen->setVisibility(btnBack,true);
534  }
535  if(page == totalPages){
536  screen->setVisibility(btnNext,false);
537  }else{
538  screen->setVisibility(btnNext,true);
539  }
540 }
