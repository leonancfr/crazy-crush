123  SS << "No record";

125  SS << "Record: "<<Game::Save::getLevelPoints(offset+i);

127  btn->setSubLabel(SS.str());
128  }
129 
130  if(actualPage == 0){
131  screenLevelSelect.setVisibility(btnLeft,false);
132  }else{
133  screenLevelSelect.setVisibility(btnLeft,true);
134  btnLeft->setOnClick(new CallbackGoToPageButtonClick(actualPage-1));

143  S="";

146  lblPage->setString(SS.str().c_str());
